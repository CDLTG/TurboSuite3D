(Revit 2025 C# Add-In)
________________________________________
1. Purpose & Scope
•	An addin where users can open a new window and change family types for Lighting Devices in the project using information gathered about the Electrical Circuit that the Lighting Devices are a part of.
________________________________________
2. Revit & API Context
•	Revit Version: 2025
•	Language: C#
•	ExternalApplication
________________________________________
4. Execution Workflow
•	User clicks the TurboDriver button on ribbon
•	New window opens which displays every Electrical Circuit in the project, filtered to include only Electrical Circuits that have a Lighting Device with a non-null value for it's Switch ID (parameter) connected. Electrical Circuits are arranged and display:
		- Circuit Number (parameter)
		- Load Name (parameter)
		- Load Classification Abbreviation (parameter)
		- Number of Elements (parameter)
		- Apparent Power (parameter)
		- Panel (parameter)
•	Embedded with each Electrical Circuit, information about the connected Lighting Fixtures is shown and arragned as follows:
		- Type Mark (parameter)
		- Linear Length (parameter) - calculate total
•	Embedded with each Electrical Circuit, information about the connected Lighting Devices is shown. This is where the user will make changes. The Lighting Devices are displayed and arranged by Switch ID (parameter) and Family Type. Users will make changes to the Family Type using a dropdown which displays all types in the project.

using Autodesk.Revit.Attributes;
using Autodesk.Revit.DB;
using System;
using System.Linq;
using System.Reflection;
using System.Windows;
using System.Windows.Controls;
using RevitUI = Autodesk.Revit.UI;

namespace TurboDriver
{
    // ================================
    // APPLICATION (Ribbon)
    // ================================
    public class App : RevitUI.IExternalApplication
    {
        public RevitUI.Result OnStartup(RevitUI.UIControlledApplication app)
        {
            string tabName = "Turbo Tools";
            try { app.CreateRibbonTab(tabName); } catch { }

            RevitUI.RibbonPanel panel = app.CreateRibbonPanel(tabName, "Driver");

            RevitUI.PushButtonData buttonData = new RevitUI.PushButtonData(
                "TurboDriver",
                "Turbo Driver",
                Assembly.GetExecutingAssembly().Location,
                "TurboDriver.TurboDriverCommand");

            panel.AddItem(buttonData);

            return RevitUI.Result.Succeeded;
        }

        public RevitUI.Result OnShutdown(RevitUI.UIControlledApplication app)
        {
            return RevitUI.Result.Succeeded;
        }
    }

    // ================================
    // COMMAND
    // ================================

    [Transaction(TransactionMode.Manual)]
    public class TurboDriverCommand : RevitUI.IExternalCommand
    {
        public RevitUI.Result Execute(
            RevitUI.ExternalCommandData commandData,
            ref string message,
            ElementSet elements)
        {
            RevitUI.UIApplication uiapp = commandData.Application;
            Document doc = uiapp.ActiveUIDocument.Document;

            TurboDriverWindow window = new TurboDriverWindow(doc);
            window.ShowDialog();

            return RevitUI.Result.Succeeded;
        }
    }

    // ================================
    // WPF WINDOW
    // ================================
    public class TurboDriverWindow : Window
    {
        private readonly Document _doc;

        public TurboDriverWindow(Document doc)
        {
            _doc = doc;

            Title = "Turbo Driver";
            Width = 300;
            Height = 200;

            StackPanel panel = new StackPanel { Margin = new Thickness(10) };

            ComboBox combo = new ComboBox { Margin = new Thickness(0, 0, 0, 10) };
            combo.ItemsSource = new FilteredElementCollector(_doc)
                .OfClass(typeof(FamilyInstance))
                .Cast<FamilyInstance>()
                .Select(x => x.Name)
                .Distinct()
                .OrderBy(x => x)
                .ToList();

            if (combo.Items.Count > 0)
                combo.SelectedIndex = 0;

            Button okButton = new Button
            {
                Content = "OK",
                Height = 30
            };

            okButton.Click += (s, e) => Close();

            panel.Children.Add(combo);
            panel.Children.Add(okButton);

            Content = panel;
        }
    }
}
