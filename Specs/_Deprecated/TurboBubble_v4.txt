using Autodesk.Revit.Attributes;
using Autodesk.Revit.DB;
using Autodesk.Revit.DB.Electrical;
using Autodesk.Revit.UI;
using Autodesk.Revit.UI.Selection;
using System;
using System.Collections.Generic;

[Transaction(TransactionMode.Manual)]
public class TurboBubble : IExternalCommand
{
    // Unit conversion - pre-computed for performance
    private const double INCHES_TO_FEET = 1.0 / 12.0;
    private const double FEET_TO_INCHES = 12.0;

    // Pre-computed default values in feet (avoid repeated multiplication)
    private const double DEFAULT_SYMBOL_SIZE_FT = 4.5 / 12.0;
    private const double MIN_SYMBOL_WIDTH_FT = 4.0 / 12.0;
    private const double TAG_WIDTH_2_CHARS_FT = 7.75 / 12.0;
    private const double TAG_WIDTH_3_CHARS_FT = 11.15 / 12.0;

    // Wire geometry offsets in feet (pre-converted to avoid runtime division)
    private const double WIRE_HORIZONTAL_OFFSET_FT = 0.613 / 12.0;
    private const double WIRE_VERTICAL_OFFSET_FT = 0.65 / 12.0;
    private const double WIRE_TO_SYMBOL_GAP_FT = 4.215 / 12.0;
    private const double WIRE_TO_SYMBOL_GAP_HORIZONTAL_FT = 4.55 / 12.0;
    private const double WIRE_ELBOW_OFFSET_FT = 3.56 / 12.0;
    private const double TAG_OFFSET_VERTICAL_FT = 5.25 / 12.0;
    private const double TAG_OFFSET_HORIZONTAL_FT = 8.75 / 12.0;

    // Rotation threshold
    private const double ROTATION_EPSILON = 0.001;

    // Required family name
    private const string SWITCHLEG_TAG_FAMILY = "AL_Tag_Lighting Fixture (Switchleg)";

    // Reusable list to avoid allocation in Wire.Create
    private static readonly List<XYZ> _wireVertices = new List<XYZ>(2);

    public Result Execute(ExternalCommandData commandData, ref string message, ElementSet elements)
    {
        UIDocument uidoc = commandData.Application.ActiveUIDocument;
        Document doc = uidoc.Document;
        View activeView = doc.ActiveView;

        try
        {
            // Fast view type check
            ViewType vt = activeView.ViewType;
            if (vt != ViewType.FloorPlan && vt != ViewType.CeilingPlan)
                return Result.Cancelled;

            // Get required element IDs upfront (before user interaction)
            ElementId tagTypeId = null;
            ElementId wireTypeId = null;

            // Single collector pass for tag types - use WhereElementIsElementType for efficiency
            using (FilteredElementCollector collector = new FilteredElementCollector(doc))
            {
                foreach (Element elem in collector.OfCategory(BuiltInCategory.OST_LightingFixtureTags)
                                                   .OfClass(typeof(FamilySymbol)))
                {
                    if (elem is FamilySymbol fs && fs.FamilyName == SWITCHLEG_TAG_FAMILY)
                    {
                        tagTypeId = fs.Id;
                        break;
                    }
                }
            }

            if (tagTypeId == null)
            {
                TaskDialog.Show("Error", $"Load tag '{SWITCHLEG_TAG_FAMILY}' before using TurboBubble.");
                return Result.Cancelled;
            }

            // Get wire type
            using (FilteredElementCollector collector = new FilteredElementCollector(doc))
            {
                wireTypeId = collector.OfCategory(BuiltInCategory.OST_Wire)
                                      .OfClass(typeof(WireType))
                                      .FirstElementId();
            }

            if (wireTypeId == null || wireTypeId == ElementId.InvalidElementId)
            {
                TaskDialog.Show("Error", "No wire types available in the project.");
                return Result.Failed;
            }

            // User selection
            Reference tagRef = uidoc.Selection.PickObject(
                ObjectType.Element,
                new LightingFixtureTagSelectionFilter(),
                "Select a lighting fixture tag");

            IndependentTag selectedTag = (IndependentTag)doc.GetElement(tagRef);

            // Get tagged fixture - use GetTaggedLocalElementIds without LINQ
            ElementId taggedId = ElementId.InvalidElementId;
            foreach (ElementId id in selectedTag.GetTaggedLocalElementIds())
            {
                taggedId = id;
                break;
            }

            if (taggedId == ElementId.InvalidElementId)
            {
                TaskDialog.Show("Error", "Tag has no associated element.");
                return Result.Failed;
            }

            FamilyInstance fixture = doc.GetElement(taggedId) as FamilyInstance;
            if (fixture == null)
            {
                TaskDialog.Show("Error", "Tagged element is not a lighting fixture.");
                return Result.Failed;
            }

            // Get electrical connector
            Connector fixtureConnector = GetElectricalConnector(fixture);
            if (fixtureConnector == null)
            {
                TaskDialog.Show("Error", "Lighting fixture has no electrical connector.");
                return Result.Failed;
            }

            // Calculate all placement data in one pass
            PlacementData data = CalculatePlacementData(doc, activeView, fixture, selectedTag);

            // Pick flip point with consolidated sketch plane transaction
            XYZ flipPoint = PickFlipPoint(uidoc, doc, data.FixturePoint);
            XYZ flipLocal = data.GlobalToLocal.OfPoint(flipPoint);

            // Determine flip state
            bool flipState = data.Condition <= 2
                ? flipLocal.Y < data.TagLocal.Y
                : flipLocal.X < data.TagLocal.X;

            // Calculate final positions
            CalculateFinalPositions(ref data, flipState);

            // Single transaction for both tag and wire creation
            using (Transaction trans = new Transaction(doc, "TurboBubble"))
            {
                trans.Start();

                // Create tag
                IndependentTag newTag = IndependentTag.Create(
                    doc, activeView.Id, new Reference(fixture), false,
                    TagMode.TM_ADDBY_CATEGORY, TagOrientation.Horizontal, data.NewTagPosition);

                newTag.ChangeTypeId(tagTypeId);
                newTag.TagHeadPosition = data.NewTagPosition;

                // Apply rotation
                if (data.RotatesWithComponent)
                {
                    // Copy orientation from selected tag
                    Parameter srcOrientation = selectedTag.LookupParameter("Orientation");
                    Parameter dstOrientation = newTag.LookupParameter("Orientation");
                    if (srcOrientation != null && dstOrientation != null && !dstOrientation.IsReadOnly)
                        dstOrientation.Set(srcOrientation.AsInteger());

                    // Rotate to match fixture
                    if (Math.Abs(data.Rotation) > ROTATION_EPSILON)
                    {
                        Line axis = Line.CreateBound(data.NewTagPosition, data.NewTagPosition + XYZ.BasisZ * 10);
                        ElementTransformUtils.RotateElement(doc, newTag.Id, axis, data.Rotation);
                    }
                }
                else
                {
                    // Set to Model orientation and apply angle
                    Parameter orientationParam = newTag.LookupParameter("Orientation");
                    if (orientationParam != null && !orientationParam.IsReadOnly)
                        orientationParam.Set(2);

                    Parameter angleParam = newTag.LookupParameter("Angle");
                    if (angleParam != null && !angleParam.IsReadOnly)
                        angleParam.Set(data.Rotation);
                }

                // Create wire - reuse static list
                _wireVertices.Clear();
                _wireVertices.Add(data.Vertex2);
                _wireVertices.Add(data.Vertex3);

                Wire.Create(doc, wireTypeId, activeView.Id, WiringType.Arc, _wireVertices, fixtureConnector, null);

                trans.Commit();
            }

            return Result.Succeeded;
        }
        catch (OperationCanceledException)
        {
            return Result.Cancelled;
        }
        catch (Exception ex)
        {
            TaskDialog.Show("Error", $"Unexpected error: {ex.Message}");
            return Result.Failed;
        }
    }

    private PlacementData CalculatePlacementData(Document doc, View view, FamilyInstance fixture, IndependentTag selectedTag)
    {
        PlacementData data = new PlacementData();

        // Get positions
        data.FixturePoint = ((LocationPoint)fixture.Location).Point;
        data.TagPoint = selectedTag.TagHeadPosition;

        // Determine if tag rotates with component (check once, store result)
        Parameter orientationParam = selectedTag.LookupParameter("Orientation");
        if (orientationParam != null)
        {
            string orientation = orientationParam.AsValueString();
            data.RotatesWithComponent = (orientation == "Horizontal" || orientation == "Vertical");
        }

        // Get rotation
        if (data.RotatesWithComponent)
        {
            data.Rotation = GetFixtureRotation(fixture);
        }
        else
        {
            Parameter angleParam = selectedTag.LookupParameter("Angle");
            data.Rotation = angleParam?.AsDouble() ?? 0.0;
        }

        // Create transforms
        data.GlobalToLocal = Transform.CreateRotationAtPoint(XYZ.BasisZ, -data.Rotation, data.FixturePoint);
        data.LocalToGlobal = data.GlobalToLocal.Inverse;

        // Transform to local
        data.FixtureLocal = data.GlobalToLocal.OfPoint(data.FixturePoint);
        data.TagLocal = data.GlobalToLocal.OfPoint(data.TagPoint);

        // Determine condition
        double diffX = data.TagLocal.X - data.FixtureLocal.X;
        double diffY = data.TagLocal.Y - data.FixtureLocal.Y;
        double absDiffX = Math.Abs(diffX);
        double absDiffY = Math.Abs(diffY);

        if (absDiffX > absDiffY)
            data.Condition = diffX > 0 ? 1 : 2;
        else
            data.Condition = diffY > 0 ? 3 : 4;

        // Get symbol dimensions from the fixture instance
        data.SymbolLength = fixture.LookupParameter("Symbol Length")?.AsDouble() ?? DEFAULT_SYMBOL_SIZE_FT;
        data.SymbolWidth = fixture.LookupParameter("Symbol Width")?.AsDouble() ?? DEFAULT_SYMBOL_SIZE_FT;

        if (data.SymbolWidth <= MIN_SYMBOL_WIDTH_FT)
            data.SymbolWidth = MIN_SYMBOL_WIDTH_FT;

        // Get tag width based on text
        string tagText = selectedTag.TagText;
        int charCount = tagText?.Length ?? 0;

        double tagWidth;
        if (charCount == 2)
            tagWidth = TAG_WIDTH_2_CHARS_FT;
        else if (charCount == 3)
            tagWidth = TAG_WIDTH_3_CHARS_FT;
        else
        {
            BoundingBoxXYZ bbox = selectedTag.get_BoundingBox(view);
            tagWidth = bbox != null ? Math.Abs(bbox.Max.X - bbox.Min.X) : 0;
        }

        if (tagWidth > data.SymbolWidth)
            data.SymbolWidth = tagWidth;

        return data;
    }

    private static double GetFixtureRotation(FamilyInstance fixture)
    {
        // Try geometry transform first
        using (Options geomOptions = new Options())
        {
            geomOptions.ComputeReferences = false;
            GeometryElement fixtureGeom = fixture.get_Geometry(geomOptions);

            if (fixtureGeom != null)
            {
                foreach (GeometryObject geomObj in fixtureGeom)
                {
                    if (geomObj is GeometryInstance geomInst)
                    {
                        XYZ xAxis = geomInst.Transform.BasisX;
                        double rotation = Math.Atan2(xAxis.Y, xAxis.X);
                        if (Math.Abs(rotation) > ROTATION_EPSILON)
                            return rotation;
                        break; // Only check first GeometryInstance
                    }
                }
            }
        }

        // Fallback to LocationPoint
        return ((LocationPoint)fixture.Location)?.Rotation ?? 0.0;
    }

    private void CalculateFinalPositions(ref PlacementData data, bool flipState)
    {
        double halfLength = data.SymbolLength * 0.5;
        double halfWidth = data.SymbolWidth * 0.5;

        double v2X, v2Y, v3X, v3Y, tagOffsetX, tagOffsetY;
        double z = data.FixtureLocal.Z;

        // Calculate offsets based on condition (all in feet now)
        switch (data.Condition)
        {
            case 1: // Right
                v2X = -WIRE_HORIZONTAL_OFFSET_FT;
                v2Y = halfLength + WIRE_TO_SYMBOL_GAP_FT;
                v3X = -WIRE_ELBOW_OFFSET_FT;
                v3Y = 0;
                tagOffsetX = 0;
                tagOffsetY = halfLength + TAG_OFFSET_VERTICAL_FT;
                break;

            case 2: // Left
                v2X = WIRE_HORIZONTAL_OFFSET_FT;
                v2Y = halfLength + WIRE_TO_SYMBOL_GAP_FT;
                v3X = WIRE_ELBOW_OFFSET_FT;
                v3Y = 0;
                tagOffsetX = 0;
                tagOffsetY = halfLength + TAG_OFFSET_VERTICAL_FT;
                break;

            case 3: // Up
                v2X = halfWidth + WIRE_TO_SYMBOL_GAP_HORIZONTAL_FT;
                v2Y = -WIRE_VERTICAL_OFFSET_FT;
                v3X = -WIRE_ELBOW_OFFSET_FT;
                v3Y = 0;
                tagOffsetX = halfWidth + TAG_OFFSET_HORIZONTAL_FT;
                tagOffsetY = data.TagLocal.Y - data.FixtureLocal.Y;
                break;

            default: // Case 4: Down
                v2X = halfWidth + WIRE_TO_SYMBOL_GAP_HORIZONTAL_FT;
                v2Y = WIRE_VERTICAL_OFFSET_FT;
                v3X = -WIRE_ELBOW_OFFSET_FT;
                v3Y = 0;
                tagOffsetX = halfWidth + TAG_OFFSET_HORIZONTAL_FT;
                tagOffsetY = data.TagLocal.Y - data.FixtureLocal.Y;
                break;
        }

        // Apply flip
        if (flipState)
        {
            if (data.Condition <= 2)
            {
                v2Y = -v2Y;
                tagOffsetY = -tagOffsetY;
            }
            else
            {
                v2X = -v2X;
                v3X = -v3X;
                tagOffsetX = -tagOffsetX;
            }
        }

        // Calculate positions in local coordinates
        XYZ newTagLocal;
        if (data.Condition <= 2)
        {
            // For left/right conditions, position tag 9" from fixture in direction of existing tag
            double tagXOffset = Math.Sign(data.TagLocal.X - data.FixtureLocal.X) * (9.0 * INCHES_TO_FEET);
            newTagLocal = new XYZ(data.FixtureLocal.X + tagXOffset, data.FixtureLocal.Y + tagOffsetY, z);
        }
        else
            newTagLocal = new XYZ(data.FixtureLocal.X + tagOffsetX, data.TagLocal.Y, z);

        XYZ vertex2Local = new XYZ(data.FixtureLocal.X + v2X, data.FixtureLocal.Y + v2Y, z);
        XYZ vertex3Local = new XYZ(newTagLocal.X + v3X, data.Condition <= 2 ? newTagLocal.Y : newTagLocal.Y + v3Y, z);

        // Transform to global
        data.NewTagPosition = data.LocalToGlobal.OfPoint(newTagLocal);
        data.Vertex2 = data.LocalToGlobal.OfPoint(vertex2Local);
        data.Vertex3 = data.LocalToGlobal.OfPoint(vertex3Local);
    }

    private XYZ PickFlipPoint(UIDocument uidoc, Document doc, XYZ origin)
    {
        View activeView = uidoc.ActiveView;
        SketchPlane originalSketchPlane = activeView.SketchPlane;
        ElementId tempSketchPlaneId = ElementId.InvalidElementId;

        try
        {
            // Create temporary sketch plane
            using (Transaction trans = new Transaction(doc, "Temp Sketch Plane"))
            {
                trans.Start();
                Plane plane = Plane.CreateByNormalAndOrigin(XYZ.BasisZ, origin);
                SketchPlane tempPlane = SketchPlane.Create(doc, plane);
                tempSketchPlaneId = tempPlane.Id;
                activeView.SketchPlane = tempPlane;
                trans.Commit();
            }

            return uidoc.Selection.PickPoint("Pick flip direction point");
        }
        finally
        {
            // Cleanup in single transaction
            using (Transaction trans = new Transaction(doc, "Cleanup"))
            {
                trans.Start();
                if (originalSketchPlane != null)
                    activeView.SketchPlane = originalSketchPlane;
                if (tempSketchPlaneId != ElementId.InvalidElementId)
                    doc.Delete(tempSketchPlaneId);
                trans.Commit();
            }
        }
    }

    private static Connector GetElectricalConnector(FamilyInstance fixture)
    {
        ConnectorSet connectors = fixture.MEPModel?.ConnectorManager?.Connectors;
        if (connectors == null) return null;

        foreach (Connector conn in connectors)
        {
            if (conn.Domain == Domain.DomainElectrical)
                return conn;
        }
        return null;
    }

    /// <summary>
    /// Struct to hold all placement calculation data - avoids heap allocation
    /// </summary>
    private struct PlacementData
    {
        public XYZ FixturePoint;
        public XYZ TagPoint;
        public XYZ FixtureLocal;
        public XYZ TagLocal;
        public Transform GlobalToLocal;
        public Transform LocalToGlobal;
        public double Rotation;
        public int Condition;
        public double SymbolLength;
        public double SymbolWidth;
        public bool RotatesWithComponent;

        // Final calculated positions
        public XYZ NewTagPosition;
        public XYZ Vertex2;
        public XYZ Vertex3;
    }
}

/// <summary>
/// Selection filter for lighting fixture tags - optimized to avoid LINQ allocations
/// </summary>
public class LightingFixtureTagSelectionFilter : ISelectionFilter
{
    private const long LIGHTING_FIXTURES_CATEGORY = (long)BuiltInCategory.OST_LightingFixtures;

    public bool AllowElement(Element elem)
    {
        IndependentTag tag = elem as IndependentTag;
        if (tag == null) return false;

        foreach (ElementId id in tag.GetTaggedLocalElementIds())
        {
            Element taggedElem = elem.Document.GetElement(id);
            if (taggedElem?.Category?.Id.Value == LIGHTING_FIXTURES_CATEGORY)
                return true;
        }
        return false;
    }

    public bool AllowReference(Reference reference, XYZ position) => false;
}
