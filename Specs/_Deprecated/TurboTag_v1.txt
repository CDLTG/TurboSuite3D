using System;
using System.Collections.Generic;
using System.Linq;
using Autodesk.Revit.Attributes;
using Autodesk.Revit.DB;
using Autodesk.Revit.UI;

[Transaction(TransactionMode.Manual)]
[Regeneration(RegenerationOption.Manual)]
public class TurboTag : IExternalCommand
{
    private const string TAG_FAMILY_NAME = "AL_Tag_Lighting Fixture (Type)";
    private const string LINEAR_TAG_FAMILY_NAME = "AL_Tag_Lighting Fixture (Linear Length)";
    private const double DEFAULT_SYMBOL_SIZE_FEET = 4.5 / 12.0;
    private const double OFFSET_MARGIN_RIGHT_FEET = 2.35 / 12.0;
    private const double OFFSET_MARGIN_LEFT_FEET = 1.7 / 12.0;
    private const double LINEAR_OFFSET_FEET = 5.0 / 12.0;
    private const double VERTICAL_OFFSET_FEET = 4.25 / 12.0;
    private const double DEFAULT_TAG_WIDTH_SHORT = 7.75 / 12.0;
    private const double DEFAULT_TAG_WIDTH_MEDIUM = 11.15 / 12.0;
    private const int SHORT_TEXT_THRESHOLD = 2;
    private const int MEDIUM_TEXT_THRESHOLD = 3;

    private static ElementId _cachedTagTypeId = ElementId.InvalidElementId;
    private static string _cachedDocumentPath;

    public Result Execute(ExternalCommandData commandData, ref string message, ElementSet elements)
    {
        UIDocument uidoc = commandData.Application.ActiveUIDocument;
        Document doc = uidoc.Document;

        try
        {
            if (!IsValidViewType(doc.ActiveView))
            {
                TaskDialog.Show("TurboTag", "Active view must be a Floor Plan or Reflected Ceiling Plan.");
                return Result.Cancelled;
            }

            var selectedFixtures = GetSelectedLightingFixtures(doc, uidoc.Selection.GetElementIds());
            if (selectedFixtures.Count == 0)
            {
                TaskDialog.Show("TurboTag", "No lighting fixtures selected.\nSelect at least one lighting fixture.");
                return Result.Cancelled;
            }

            // Separate fixtures into line-based and point-based
            var lineBasedFixtures = selectedFixtures.Where(f => IsLineBased(f)).ToList();
            var pointBasedFixtures = selectedFixtures.Where(f => !IsLineBased(f)).ToList();

            int totalTagged = 0;

            // Process line-based fixtures
            if (lineBasedFixtures.Count > 0)
            {
                TagDirection linearDirection = PromptForDirectionLinear();
                if (linearDirection == TagDirection.None)
                {
                    return Result.Cancelled;
                }

                string linearTypeName = linearDirection == TagDirection.Up ? "Tag_Top" : "Tag_Bottom";
                FamilySymbol linearTagType = GetLinearTagType(doc, linearTypeName);
                if (linearTagType == null)
                {
                    TaskDialog.Show("TurboTag", $"Tag type '{linearTypeName}' in family '{LINEAR_TAG_FAMILY_NAME}' not found.\nLoad this tag family into the project.");
                    return Result.Cancelled;
                }

                totalTagged += PlaceTags(doc, lineBasedFixtures, linearTagType, linearDirection, true);
            }

            // Process point-based fixtures
            if (pointBasedFixtures.Count > 0)
            {
                FamilySymbol tagType = GetTagType(doc);
                if (tagType == null)
                {
                    TaskDialog.Show("TurboTag", $"Tag family '{TAG_FAMILY_NAME}' not found.\nLoad this tag family into the project.");
                    return Result.Cancelled;
                }

                TagDirection direction = PromptForDirection();
                if (direction == TagDirection.None)
                {
                    return Result.Cancelled;
                }

                totalTagged += PlaceTags(doc, pointBasedFixtures, tagType, direction);
            }

            if (selectedFixtures.Count > 10)
            {
                TaskDialog.Show("TurboTag", $"Successfully tagged {totalTagged} of {selectedFixtures.Count} fixtures.");
            }

            return Result.Succeeded;
        }
        catch (Autodesk.Revit.Exceptions.OperationCanceledException)
        {
            return Result.Cancelled;
        }
        catch (Exception ex)
        {
            message = ex.Message;
            TaskDialog.Show("TurboTag Error", $"An unexpected error occurred:\n{ex.Message}");
            return Result.Failed;
        }
    }

    private static bool IsLineBased(FamilyInstance fixture)
    {
        return fixture.Location is LocationCurve;
    }

    private static bool IsValidViewType(View view)
    {
        return view.ViewType == ViewType.FloorPlan || view.ViewType == ViewType.CeilingPlan;
    }

    private static List<FamilyInstance> GetSelectedLightingFixtures(Document doc, ICollection<ElementId> selectedIds)
    {
        if (selectedIds.Count == 0)
            return new List<FamilyInstance>();

        var lightingCategoryId = new ElementId(BuiltInCategory.OST_LightingFixtures);
        var fixtures = new List<FamilyInstance>(selectedIds.Count);

        foreach (ElementId id in selectedIds)
        {
            if (doc.GetElement(id) is FamilyInstance fi &&
                fi.Category?.Id == lightingCategoryId)
            {
                fixtures.Add(fi);
            }
        }

        return fixtures;
    }

    private FamilySymbol GetTagType(Document doc)
    {
        string currentPath = doc.PathName ?? doc.Title;

        if (_cachedTagTypeId != ElementId.InvalidElementId &&
            _cachedDocumentPath == currentPath)
        {
            var cached = doc.GetElement(_cachedTagTypeId) as FamilySymbol;
            if (cached != null && cached.IsValidObject)
                return cached;
        }

        var tagType = new FilteredElementCollector(doc)
            .OfClass(typeof(FamilySymbol))
            .OfCategory(BuiltInCategory.OST_LightingFixtureTags)
            .Cast<FamilySymbol>()
            .FirstOrDefault(fs => string.Equals(fs.FamilyName, TAG_FAMILY_NAME, StringComparison.OrdinalIgnoreCase));

        if (tagType != null)
        {
            _cachedTagTypeId = tagType.Id;
            _cachedDocumentPath = currentPath;
        }

        return tagType;
    }

    private FamilySymbol GetLinearTagType(Document doc, string typeName)
    {
        string currentPath = doc.PathName ?? doc.Title;
        string cacheKey = $"{currentPath}_{typeName}";

        var tagType = new FilteredElementCollector(doc)
            .OfClass(typeof(FamilySymbol))
            .OfCategory(BuiltInCategory.OST_LightingFixtureTags)
            .Cast<FamilySymbol>()
            .FirstOrDefault(fs => string.Equals(fs.FamilyName, LINEAR_TAG_FAMILY_NAME, StringComparison.OrdinalIgnoreCase)
                               && string.Equals(fs.Name, typeName, StringComparison.OrdinalIgnoreCase));

        return tagType;
    }

    private static TagDirection PromptForDirectionLinear()
    {
        var dialog = new TaskDialog("Tag Direction")
        {
            MainInstruction = "Select tag placement direction for line-based families:",
            CommonButtons = TaskDialogCommonButtons.Cancel
        };

        dialog.AddCommandLink(TaskDialogCommandLinkId.CommandLink1, "Up");
        dialog.AddCommandLink(TaskDialogCommandLinkId.CommandLink2, "Down");

        return dialog.Show() switch
        {
            TaskDialogResult.CommandLink1 => TagDirection.Up,
            TaskDialogResult.CommandLink2 => TagDirection.Down,
            _ => TagDirection.None
        };
    }

    private static TagDirection PromptForDirection()
    {
        var dialog = new TaskDialog("Tag Direction")
        {
            MainInstruction = "Select tag placement direction for point-based families:",
            CommonButtons = TaskDialogCommonButtons.Cancel
        };

        dialog.AddCommandLink(TaskDialogCommandLinkId.CommandLink1, "Up");
        dialog.AddCommandLink(TaskDialogCommandLinkId.CommandLink2, "Down");
        dialog.AddCommandLink(TaskDialogCommandLinkId.CommandLink3, "Right");
        dialog.AddCommandLink(TaskDialogCommandLinkId.CommandLink4, "Left");

        return dialog.Show() switch
        {
            TaskDialogResult.CommandLink1 => TagDirection.Up,
            TaskDialogResult.CommandLink2 => TagDirection.Down,
            TaskDialogResult.CommandLink3 => TagDirection.Right,
            TaskDialogResult.CommandLink4 => TagDirection.Left,
            _ => TagDirection.None
        };
    }

    private int PlaceTags(Document doc, List<FamilyInstance> fixtures, FamilySymbol tagType, TagDirection direction, bool isLineBased = false)
    {
        int successCount = 0;
        View activeView = doc.ActiveView;
        ElementId tagTypeId = tagType.Id;
        ElementId viewId = activeView.Id;

        using (var trans = new Transaction(doc, "TurboTag - Place Tags"))
        {
            var failureOptions = trans.GetFailureHandlingOptions();
            failureOptions.SetFailuresPreprocessor(new TagFailurePreprocessor());
            trans.SetFailureHandlingOptions(failureOptions);

            trans.Start();

            foreach (FamilyInstance fixture in fixtures)
            {
                if (TryPlaceTag(doc, fixture, tagTypeId, viewId, direction, isLineBased))
                {
                    successCount++;
                }
            }

            trans.Commit();
        }

        return successCount;
    }

    private bool TryPlaceTag(Document doc, FamilyInstance fixture, ElementId tagTypeId, ElementId viewId, TagDirection direction, bool isLineBased = false)
    {
        try
        {
            XYZ fixtureLocation;

            if (isLineBased)
            {
                if (!(fixture.Location is LocationCurve locCurve))
                    return false;

                // Use midpoint of line-based fixture
                Curve curve = locCurve.Curve;
                fixtureLocation = curve.Evaluate(0.5, true);
            }
            else
            {
                if (!(fixture.Location is LocationPoint locPoint))
                    return false;

                fixtureLocation = locPoint.Point;
            }

            var reference = new Reference(fixture);
            IndependentTag tag = IndependentTag.Create(
                doc, tagTypeId, viewId, reference,
                addLeader: false,
                TagOrientation.Horizontal,
                fixtureLocation);

            if (tag == null)
                return false;

            XYZ localOffset;

            if (isLineBased)
            {
                localOffset = CalculateLinearOffset(direction);
            }
            else
            {
                double symbolLength = GetParameterValueOrDefault(fixture, "Symbol Length", DEFAULT_SYMBOL_SIZE_FEET);
                double symbolWidth = GetParameterValueOrDefault(fixture, "Symbol Width", DEFAULT_SYMBOL_SIZE_FEET);
                double tagWidth = EstimateTagWidth(tag, doc, viewId);
                localOffset = CalculateOffset(direction, symbolLength, symbolWidth, tagWidth);
            }

            XYZ globalOffset = TransformToGlobal(fixture, localOffset);

            if (!globalOffset.IsZeroLength())
            {
                ElementTransformUtils.MoveElement(doc, tag.Id, globalOffset);
            }

            return true;
        }
        catch (Autodesk.Revit.Exceptions.InvalidOperationException)
        {
            return false;
        }
        catch
        {
            return false;
        }
    }

    private static XYZ TransformToGlobal(FamilyInstance fixture, XYZ localOffset)
    {
        Transform transform = fixture.GetTransform();

        XYZ globalOffset = (transform.BasisX * localOffset.X) +
                           (transform.BasisY * localOffset.Y) +
                           (transform.BasisZ * localOffset.Z);

        return globalOffset;
    }

    private static XYZ CalculateOffset(TagDirection direction, double symbolLength, double symbolWidth, double tagWidth)
    {
        return direction switch
        {
            TagDirection.Up => new XYZ(0, -((symbolLength / 2.0) + VERTICAL_OFFSET_FEET), 0),
            TagDirection.Down => new XYZ(0, (symbolLength / 2.0) + VERTICAL_OFFSET_FEET, 0),
            TagDirection.Right => new XYZ((symbolWidth / 2.0) + (tagWidth / 2.0) + OFFSET_MARGIN_RIGHT_FEET, 0, 0),
            TagDirection.Left => new XYZ(-((symbolWidth / 2.0) + (tagWidth / 2.0) + OFFSET_MARGIN_LEFT_FEET), 0, 0),
            _ => XYZ.Zero
        };
    }

    private static XYZ CalculateLinearOffset(TagDirection direction)
    {
        return direction switch
        {
            TagDirection.Up => new XYZ(0, -LINEAR_OFFSET_FEET, 0),
            TagDirection.Down => new XYZ(0, LINEAR_OFFSET_FEET, 0),
            _ => XYZ.Zero
        };
    }

    private static double GetParameterValueOrDefault(FamilyInstance element, string parameterName, double defaultValue)
    {
        Parameter param = element.LookupParameter(parameterName);
        return (param != null && param.HasValue) ? param.AsDouble() : defaultValue;
    }

    private static double EstimateTagWidth(IndependentTag tag, Document doc, ElementId viewId)
    {
        try
        {
            string tagText = tag.TagText;

            if (string.IsNullOrEmpty(tagText) || tagText.Length <= SHORT_TEXT_THRESHOLD)
            {
                return DEFAULT_TAG_WIDTH_SHORT;
            }

            if (tagText.Length == MEDIUM_TEXT_THRESHOLD)
            {
                return DEFAULT_TAG_WIDTH_MEDIUM;
            }

            View view = viewId != ElementId.InvalidElementId ? doc.GetElement(viewId) as View : null;
            BoundingBoxXYZ bbox = tag.get_BoundingBox(view);

            if (bbox != null)
            {
                return bbox.Max.X - bbox.Min.X;
            }

            return DEFAULT_TAG_WIDTH_MEDIUM;
        }
        catch
        {
            return DEFAULT_TAG_WIDTH_SHORT;
        }
    }
}

public class TagFailurePreprocessor : IFailuresPreprocessor
{
    public FailureProcessingResult PreprocessFailures(FailuresAccessor failuresAccessor)
    {
        var failures = failuresAccessor.GetFailureMessages();

        foreach (FailureMessageAccessor failure in failures)
        {
            if (failure.GetSeverity() == FailureSeverity.Warning)
            {
                failuresAccessor.DeleteWarning(failure);
            }
        }

        return FailureProcessingResult.Continue;
    }
}

public enum TagDirection
{
    None = 0,
    Up,
    Down,
    Left,
    Right
}
